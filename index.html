BMI Calculator

This tool is based on standard WHO health categories (as seen on Calculator.net).

function calculateBMI() { // 1. Get values from the HTML inputs const weight = parseFloat(document.getElementById('weight').value); const height = parseFloat(document.getElementById('height').value); const resultDiv = document.getElementById('bmi-result'); // 2. Validation if (isNaN(weight) || isNaN(height) || weight <= 0 || height <= 0) { alert("Please enter valid positive numbers for both weight and height."); return; } // 3. Calculation (Metric: kg / m^2) const heightInMeters = height / 100; const bmi = (weight / (heightInMeters * heightInMeters)).toFixed(1); // 4. Determine Category using If Statements (Based on Calculator.net / WHO) let category = ""; let color = ""; if (bmi < 18.5) { category = "Underweight"; color = "#31708f"; // Blue } else if (bmi >= 18.5 && bmi < 25) { category = "Normal"; color = "#3c763d"; // Green } else if (bmi >= 25 && bmi < 30) { category = "Overweight"; color = "#8a6d3b"; // Yellow/Brown } else { category = "Obese"; color = "#a94442"; // Red } // 5. Output Actions // Action 1: Show an alert with the result [Requirement] alert("BMI Calculation Complete!\n\nYour BMI: " + bmi + "\nCategory: " + category); // Action 2: Update the page content with styling similar to Calculator.net resultDiv.style.display = "block"; resultDiv.style.backgroundColor = color; resultDiv.style.color = "white"; resultDiv.innerHTML = "BMI = " + bmi + " kg/m2 (" + category + ")"; }
